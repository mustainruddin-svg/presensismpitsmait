Copy Web App URL → paste ke GAS_URL di file HTML // ============================================================ const SPREADSHEET_ID = '1UTf9V2-inEn1S0yAF_mZ185ddVPn9Gxw1hNopRTToUw'; const SHEET_PRESENSI = 'Presensi'; // nama sheet untuk data absen const SHEET_CONFIG = 'Config'; // nama sheet untuk konfigurasi jam // ── doGet: handle SEMUA request GET dari browser ──────────── // Browser kirim GET (no-cors), bukan POST — karena POST JSON // memerlukan CORS preflight yang diblokir Google Script. function doGet(e) { const action = e.parameter.action || ''; // === Request absen dari scanner === if (action === 'absen') { return handleAbsen(e.parameter); } // === Default: kembalikan konfigurasi jam (untuk fetchJamConfig) === return handleGetJam(); } // ── Handler: simpan data absen ke sheet ───────────────────── function handleAbsen(p) { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); let sheet = ss.getSheetByName(SHEET_PRESENSI); // Buat sheet jika belum ada if (!sheet) { sheet = ss.insertSheet(SHEET_PRESENSI); // Header row sheet.appendRow([ 'Timestamp', 'Scan QR Code Anda', 'Jam Datang', 'Sesi', 'Status', 'Nama Guru', 'Tanggal Scan', 'Jenis Hari' ]); // Format header const header = sheet.getRange(1, 1, 1, 8); header.setBackground('#1a5276'); header.setFontColor('#ffffff'); header.setFontWeight('bold'); } const now = new Date(); const timestamp = Utilities.formatDate(now, 'Asia/Makassar', 'dd/MM/yyyy HH:mm:ss'); sheet.appendRow([ timestamp, p.qrCode || '', p.jamScan || '', p.sesi || '', p.status || '', p.namaGuru || p.qrCode || '', p.tanggal || '', p.jenisHari || '', ]); return ContentService .createTextOutput(JSON.stringify({ ok: true, ts: timestamp })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── Handler: kembalikan konfigurasi jam dari sheet Config ─── function handleGetJam() { try { const ss = SpreadsheetApp.openById(SPREADSHEET_ID); const sheet = ss.getSheetByName(SHEET_CONFIG); if (!sheet) { // Kembalikan default jika sheet Config belum ada return ContentService.createTextOutput(JSON.stringify({ ok: true, jam: { jamMasuk: '08:00', jamPulang: '14:00', jamSabtu: '08:00', jamPiket: '08:00', csDatang: '06:45', csPulang: '15:00', csSabtu: '07:15', } })).setMimeType(ContentService.MimeType.JSON); } // Baca dari sheet Config (kolom A = key, kolom B = value) const data = sheet.getDataRange().getValues(); const jam = {}; data.forEach(row => { if (row[0] && row[1]) jam[row[0].toString().trim()] = row[1].toString().trim(); }); return ContentService .createTextOutput(JSON.stringify({ ok: true, jam })) .setMimeType(ContentService.MimeType.JSON); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // ── doPost: tetap ada untuk kompatibilitas mundur ──────────── function doPost(e) { try { const payload = JSON.parse(e.postData.contents); return handleAbsen(payload); } catch(err) { return ContentService .createTextOutput(JSON.stringify({ ok: false, error: err.message })) .setMimeType(ContentService.MimeType.JSON); } }
